(+ 1 2)

; テスト2: 変数の定義と参照
(define a 10)
a

; テスト3: 条件分岐
(if (= a 10) "Yes" "No")

; テスト4: 関数の定義と呼び出し
(define (add x y) (+ x y))
(add 5 3)

; テスト5: レキシカルスコープ
(let ((a 20))
  (add a 5))

; テスト6: リスト操作
(define lst (list 1 2 3 4 5))
(car lst)
(cdr lst)

; テスト7: 再帰関数
(define (factorial n)
  (if (= n 0) 1 (* n (factorial (- n 1)))))
(factorial 5)

; テスト8: クロージャ
(define (make-adder x)
  (lambda (y) (+ x y)))
(define add5 (make-adder 5))
(add5 3)
